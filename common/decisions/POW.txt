 		icon = GFX_idea_chi_army_corruption

 		}
 		available={
 			custom_trigger_tooltip= {
 				tooltip=[POW_recruit_soldiers_tt1]
 				if={
 					limit={check_variable={POW_recruit_soldiers_cost>0}}
 					check_variable ={total_pows_unassigned>POW_recruit_soldiers_cost}
 				}
 				else ={
 					check_variable ={total_pows_unassigned>25}}
 			}
 		}
 		fixed_random_seed = no
 		fire_only_once = no
 		is_good = yes
 		complete_effect = {
 			custom_effect_tooltip = [POW_recruit_soldiers_tt2]
			add_to_variable={total_pows_assigned=10}
 			subtract_from_variable={total_pows_unassigned =10}
 			}
		remove_effect = {
			random_list= {
				30 = {
 					add_manpower = 10000
 				}
				30 = {
 					add_manpower = 7500
 				}
 				30 = {
 					add_manpower = 5000
 				}
 				10 = {
 					if= {limit={has_idea=mutiny_3}
 						modify_timed_idea = {
 							idea = mutiny_3
 							days = 30
 						}
 					}
 					if= {limit={has_idea=mutiny_2}
 						remove_ideas=mutiny_2
 						add_timed_idea = {
 							idea = mutiny_3
 							days = 30
 						}
 					}
 					else_if= {limit={has_idea=mutiny_1}
 						remove_ideas=mutiny_1
 						add_timed_idea = {
 							idea = mutiny_2
 							days = 30
 						}
 					}
 					else= {
 						add_timed_idea = {
 							idea = mutiny_1
 							days = 30
 						}
 					}
 				}
		}
 		}
